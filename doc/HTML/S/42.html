/* List of Cells 
 * head: front cell
 * tail: back cell
 */ 
class List{
/* [<][>][^][v][top][bottom][index][help] */

    //inner classes
    class Cell {
/* [<][>][^][v][top][bottom][index][help] */
        public Node elm;
        public Cell next;

        public Cell(){
        }
    }

    Cell head = null;
    Cell tail = null;
    int elements = 0;

    Cell c = null;
    Node current = null;

    public void scan(List listNew){
/* [<][>][^][v][top][bottom][index][help] */
        c = head;
            while(c != null){
                current = c.elm;
                current.makeChildren();
                listNew.add3(current.child1, current.child2, current.child3);
                c = c.next;
            }
        }

    public void add(Node elm){
/* [<][>][^][v][top][bottom][index][help] */
        Cell c = new Cell();
        c.elm = elm;
        if(elements == 0){
            head = c;
            tail = c;
            elements = 1;
        } else {
            tail.next = c;
            tail = tail.next;
            elements++;
        }
    }
    
    public void add3(Node elm1, Node elm2, Node elm3){
/* [<][>][^][v][top][bottom][index][help] */
        add(elm1);
        add(elm2);
        add(elm3);
    }
    
    public void makeEmpty(){
/* [<][>][^][v][top][bottom][index][help] */
        elements = 0;
        tail = head;
    }
    
    public boolean isEmpty(){
/* [<][>][^][v][top][bottom][index][help] */
        if(elements==0){
            return true;
        } else {
            return false;
        }
    }
}
